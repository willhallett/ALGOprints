class Structure {
  color[] colors = new color[9];
  color[] plt = new color[10];
  boolean tr = true;
  float x, y;
  int[] a, b;
  int p, m;
  float[] c, d;
  float v, v2;

  Structure(float x, float y, int p, int m, float v, float v2) {
    this.x = x;
    this.y = y;
    this.p = p;
    this.m = m;
    this.v = v;
    this.v2 = v2;
    plt[0] = color(222, 69, 0);
    plt[1] = color(235, 217, 143);
    plt[2] = color(0, 62, 131);
    plt[3] = color(0, 89, 46);
    plt[4] = color(50);
    plt[5] = color(111, 0, 67);
    plt[6] = color(209, 188, 25);
    plt[7] = color(79, 142, 230);
    plt[8] = color(155, 83, 72);
    plt[9] = color(181, 209, 204);
  }

  void configure() {
    for (int i = 0; i < colors.length; i++) {
      int a = floor(random(0, plt.length));
      colors[i] = plt[a];
    }

    a = new int[m];
    b = new int[m];

    int[] r = {-1, 0, 1};

    for (int i = 0; i < m; i++) {
      int j = floor(random(0, r.length));
      int k = floor(random(0, r.length));
      a[i] = r[j];
      b[i] = r[k];
    }

    float[] del = new float[m];
    c = new float[m];
    d = new float[m];
    for (int i = 0; i < m; i++) {
      int j = floor(random(0, m));
      int k = floor(random(0, m));
      c[i] = del[j];
      d[i] = del[k];
    }
  }

  int t = 0;
  int k = 0;

  void display() {
    if (t % p == 0) {
      k++;
    }
    if (tr == true) {
      y += a[k % a.length];
      x += b[k % b.length];
    } else {
      y += c[k % c.length] * v;
      x += d[k % d.length] * v;
    }
    point(x, y);
    stroke(colors[t % colors.length]);

    if (t >= m) {
      t = 0;
      tr = !tr;
      for (int i = 0; i < colors.length; i++) {
        colors[i] = plt[floor(random(0, plt.length))];
      }
    }
    t++;
  }

  void run() {
    if (struct.x > z && struct.x < width - z && struct.y > z && struct.y < height - z) {
      struct.display();
    }
  }
}
